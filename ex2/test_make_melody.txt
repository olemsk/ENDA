int counter = 0; 




void control_music(uint32_t song)
{
const uint32_t song; // vektor med sangene
const int songlength; // må lages i sound filene

switch(song)
{
case 0xFD:
song = Beep1;
songlength = Beep1_length;


break;
case 0xFB:
song = Beep2;
songlength = Beep2_length;

break;

default:
song = Beep1;
songlength= Beep1_length;

return;
}

if ( counter == songlength -1)
{
	counter = 0;
	return;

}
else

{
*DAC0_CH0DATA = song[counter] ;
*DAC0_CH1DATA = song[counter] ;
counter++;	

}


}





bool play_beep_1(void)
{
int length = 7585;
if(counter == length)
{
*DAC0_CH0DATA = 0 ;
*DAC0_CH1DATA = 0;
counter=0;
return true;
}
uint16_t temp= Beep1[counter];
*DAC0_CH0DATA = temp ;
*DAC0_CH1DATA = temp;
counter +=1;
return false;
};
bool play_beep_2(void)
{

return true;
}

*DAC0_CH0DATA = temp ;
*DAC0_CH1DATA = temp;
counter +=1;
return false;
};
